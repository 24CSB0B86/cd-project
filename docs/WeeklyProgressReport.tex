\section{Phase 1: Problem Understanding \& Basic Foundations (Weeks 1--2)}
%% ─────────────────────────────────────────────────────────

\subsection{Week 1 --- Problem Definition \& EBNF Grammar Design}

\textbf{Status:} \done \hfill \textbf{Date Completed:} January 2026

\textbf{Planned Objectives:}
\begin{itemize}[leftmargin=*, itemsep=1pt]
    \item Define the Extended Backus-Naur Form (EBNF) grammar for Mini-Python
    \item Map the ``Agentic'' interaction flow (ReAct: Reason + Act pattern)
    \item Establish repository structure and development environment
\end{itemize}

\textbf{Achieved:}
\begin{itemize}[leftmargin=*, itemsep=1pt]
    \item Designed the Mini-Python EBNF with \textbf{42 production rules} covering: variable assignments, arithmetic expressions (5 precedence levels), \texttt{if}/\texttt{elif}/\texttt{else}, \texttt{while}/\texttt{for} loops, function definitions, \texttt{return} statements
    \item Created agentic interaction flow diagram (LucidChart)
    \item Initialized repository structure: \texttt{src/}, \texttt{tests/}, \texttt{docs/}, \texttt{reports/}, \texttt{grammar/}
\end{itemize}

\textbf{Key Deliverables:} \texttt{grammar/mini\_python.ebnf}, \texttt{reports/week\_01\_report.tex}

\textbf{Key Decision:} Chose a Python \textit{subset} (Mini-Python) rather than full Python to keep the compiler scope academically tractable while still covering the most common beginner error patterns.

\subsection{Week 2 --- Gemini API \& Agentic Workflow Design}

\textbf{Status:} \done \hfill \textbf{Date Completed:} January 2026

\textbf{Planned Objectives:}
\begin{itemize}[leftmargin=*, itemsep=1pt]
    \item Set up Google AI Studio; configure \texttt{python-dotenv} API key management
    \item Test Gemini's ability to parse token streams and JSON-serialized ASTs
    \item Design the three-tier prompt architecture
\end{itemize}

\textbf{Achieved:}
\begin{itemize}[leftmargin=*, itemsep=1pt]
    \item Configured \texttt{GEMINI\_API\_KEY} via \texttt{.env} file; confirmed \texttt{gemini-1.5-pro} access
    \item Ran 15 prompt experiments in Jupyter; found that structured JSON context injection outperforms raw source code in accuracy by $\approx$30 percentage points
    \item Designed Auditor / Assistant / Security agent prompt hierarchy

\textbf{Key Deliverables:} \texttt{docs/prompts.md}, \texttt{reports/week\_02\_report.tex}

\textbf{Key Decision:} Adopted offline-first design --- all core compiler operations are API-free, ensuring tests work without an API key. Gemini is always an optional enhancement.

\textbf{Status:} \done \hfill \textbf{Tests:} 13 $\;|\;$ \textbf{Cumulative:} 13

\textbf{Achieved:}
\begin{itemize}[leftmargin=*, itemsep=1pt]
    \item Built a Regex-based Lexer from scratch; 37 keyword types recognized
    \item Implemented Python-specific INDENT/DEDENT emission via an indentation stack
    \item Handles: line continuation, multi-line strings, inline comments
    \item 13 deterministic unit tests passing (no API calls)

\subsection{Week 13 --- Explainability \& Traceability (XAI)}

\textbf{Status:} \done \hfill \textbf{Tests:} 33 $\;|\;$ \textbf{Cumulative:} 290
